    leia(n1)
    escreva("Digite o segundo número: ")
    leia(n2)
    escreva("Digite o terceiro número: ")
    leia(n3)

    // Descobrindo o maior

        meio = n3
        menor = n2
      }

      maior = n2

        meio = n1
        menor = n3
      } senao {
        meio = n3
        menor = n1
      }

      maior = n3

        meio = n1

        meio = n2
        menor = n1
      }
    }
